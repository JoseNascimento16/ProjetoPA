lookups: lt, lte, gte (less than, greater than)

#gte
# Filtra pelos objetos mais recentes (com até 7 dias desde a ultima modificação)
planos = Plano_de_acao.objects.filter(ultima_modificacao__gte=date.today() - timedelta(days=7))

#lte
# Filtra pelos objetos mais velhos do que 7 dias desde a ultima modificação
planos = Plano_de_acao.objects.filter(ultima_modificacao__lte=date.today() - timedelta(days=7))
