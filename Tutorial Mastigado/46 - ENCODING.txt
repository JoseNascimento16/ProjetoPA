Alguns tipos de ENCODING pelos quais já quebrei a cabeça:

from django.utils.encoding import force_bytes, force_str

forcebytes_uid = force_bytes(user.id) # Converte INT para BYTES
bytes_to_b64 = base64.b64encode(forcebytes_uid) # Converte BYTES para BASE64
b64_to_b64urlsafe = urlsafe_base64_encode(base64EncodedStr) # Converte BASE64 para URLSAFE_BASE64

b64urlsafe_to_b64 = urlsafe_base64_decode(uidb64_urlsafe) # Converte URLBASE64 para BASE64
b64_to_bytes = base64.b64decode(b64urlsafe_to_b64) # Converte BASE64 para BYTES
bytes_to_str = force_str(b64_to_bytes) # Converte BYTES para STR
ou
bytes_to_int = int(force_str(b64_to_bytes)) # Converte BYTES para INT
